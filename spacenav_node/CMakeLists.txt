cmake_minimum_required(VERSION 2.4.6)
include($ENV{ROS_ROOT}/core/rosbuild/rosbuild.cmake)
rosbuild_init()
rosbuild_add_executable(spacenav_node src/spacenav_node.cpp)

find_path(libspnav_include_dir spnav.h)
if (NOT libspnav_include_dir)
 message ("LIBRARY ERROR: Looking for spnav.h - not found, please run rosdep install spacenav_node")
endif (NOT libspnav_include_dir)
include_directories(${libspnav_include_dir})
find_library(libspnav_library spnav) 
target_link_libraries (spacenav_node ${libspnav_library} X11)
